130 SDL_Event event;
131 SDL_PollEvent(&event);

133 switch(event.type) {
134 case SDL_QUIT:
135 this->is_running = false;
136 break;
137 case SDL_TEXTINPUT:
138 if(!(this->state == GAME && !this->game_screen->is_ready)) break;
139 if(event.text.text[0] == ' ') break;
140
141 if(this->game_screen->player_name_inputs[0].is_active) {

148 }
149 }
150 break;
151 case SDL_TEXTEDITING:
152 DBG_LOG("TEXT EDITING EVENT");
153 break;
154 case SDL_KEYDOWN:
155 switch(event.key.keysym.sym) {
156 case SDLK_ESCAPE:
157 this->is_running = false;

159 case SDLK_BACKSPACE:
160 // TODO: ABSOLUTELY NOT, and handle utf8 characters

205 if(this->game_screen->player_name_inputs[0].is_active) break;

486{
487 DBG_LOG("type: %d, n: %u, e: %u, s: %u, w: %u",
488 this->game_screen->drawn_tile->type,
489 this->game_screen->drawn_tile->connections[NORTH],
490 this->game_screen->drawn_tile->connections[EAST],
491 this->game_screen->drawn_tile->connections[SOUTH],
492 this->game_screen->drawn_tile->connections[WEST]);
493
494 int x = tcoords.x;

496
497 // TODO: for loop
498 // még egy korábbi refaktor előtti állapotból maradt így, amikor nem lehetett volna ciklussal megoldani
499 bool next_to_placed = false;
500 if(y - 1 >= 0) {
501 if(this->game_screen->board[x][y-1].type != EMPTY) next_to_placed = true;
502 if(this->game_screen->board[x][y-1].connections[SOUTH] != this->game_screen->drawn_tile->connections[NORTH]
503 && this->game_screen->board[x][y-1].connections[SOUTH] != NONE) {
504 return false;
505 }
506 }
507 if(y + 1 < BOARD_SIZE) {
508 if(this->game_screen->board[x][y+1].type != EMPTY) next_to_placed = true;
509 if(this->game_screen->board[x][y+1].connections[NORTH] != this->game_screen->drawn_tile->connections[SOUTH]
510 && this->game_screen->board[x][y+1].connections[NORTH] != NONE) {
511 return false;
512 }
513 }
514 if(x - 1 >= 0) {
515 if(this->game_screen->board[x-1][y].type != EMPTY) next_to_placed = true;
516 if(this->game_screen->board[x-1][y].connections[EAST] != this->game_screen->drawn_tile->connections[WEST]
517 && this->game_screen->board[x-1][y].connections[EAST] != NONE) {
518 return false;

522 if(this->game_screen->board[x+1][y].type != EMPTY) next_to_placed = true;
