258void Carcassone__handle_input(Carcassone* this)
259{
260 SDL_Event event;
261 SDL_PollEvent(&event);
262
263 switch(event.type) {
264 case SDL_QUIT:
265 this->is_running = false;
266 break;
267 case SDL_KEYDOWN:
268 if(this->state != GAME) break;
269
270 switch(event.key.keysym.sym) {
271 case SDLK_ESCAPE:
272 this->is_running = false;
273 break;
274 case SDLK_d:
275 Carcassone__draw_new(this);
276 break;
277 case SDLK_r:
278 Tile__rotate(this->drawn_tile);
279 break;
280 default:
281 Carcassone__move_board(this, event.key.keysym.sym);
282 break;
283 }
284 break;
285 case SDL_MOUSEBUTTONDOWN:
286 if(this->state == MENU) {
287 if(SDL_PointInRect(&(SDL_Point){event.button.x, event.button.y}, &this->menu->start_button.rect)) {
288 this->state = GAME;
289 }
290 if(SDL_PointInRect(&(SDL_Point){event.button.x, event.button.y}, &this->menu->lboard_button.rect)) {
291 this->state = LBOARD;
292 }
293 }
294
295 if(this->state == LBOARD) {
296 if(SDL_PointInRect(&(SDL_Point){event.button.x, event.button.y}, &this->lboard->back_button.rect)) {
297 this->state = MENU;
298 }
299 }
300
301 if(this->state != GAME) break;
302
303 for(int y = 0; y < GRID_SIZE; ++y) {
304 for(int x = 0; x < GRID_SIZE; ++x) {
305 Tile* curr_tile = &this->board[x][y];
306 if(Tile__point_in_tile(curr_tile, (SDL_Point){event.button.x, event.button.y})
307 && curr_tile->type == EMPTY
308 && Carcassone__check_surrounding_tiles(this, (SDL_Point){x, y})
309 ) {
310 // TODO
311 SDL_Point temp_new_local = {curr_tile->local_coords.x / TILE_SIZE,
312 curr_tile->local_coords.y / TILE_SIZE};
313 Tile__construct(curr_tile, this->drawn_tile->type,
314 temp_new_local, this->map_offset);
315 Tile__set_rotation(curr_tile, this->drawn_tile->rotation);
316
317 Carcassone__draw_new(this);
318
319 ++this->curr_player_index;
320 this->curr_player_index %= 2;
321 DBG_LOG("Curr player: %hu", this->curr_player_index + 1);
322 }
323 }
324 }

512bool Carcassone__check_surrounding_tiles(Carcassone* this, SDL_Point tcoords)
513{
514 int x = tcoords.x;
515 int y = tcoords.y;
516
517 // TODO: for loop
518 bool next_to_placed = false;
519 if(y - 1 >= 0) {
520 if(this->board[x][y-1].type != EMPTY) next_to_placed = true;
521 if(this->board[x][y-1].south != this->drawn_tile->north && this->board[x][y-1].south != NONE) {
522 return false;
523 }
524 }
525 if(y + 1 < GRID_SIZE) {
526 if(this->board[x][y+1].type != EMPTY) next_to_placed = true;
527 if(this->board[x][y+1].north != this->drawn_tile->south && this->board[x][y+1].north != NONE) {
528 return false;
529 }
530 }
531 if(x - 1 >= 0) {
532 if(this->board[x-1][y].type != EMPTY) next_to_placed = true;
533 if(this->board[x-1][y].east != this->drawn_tile->west && this->board[x-1][y].east != NONE) {
534 return false;
535 }
536 }
537 if(x + 1 < GRID_SIZE) {
538 if(this->board[x+1][y].type != EMPTY) next_to_placed = true;
539 if(this->board[x+1][y].west != this->drawn_tile->east && this->board[x+1][y].west != NONE) {
540 return false;
541 }
542 }
543
544 return next_to_placed;
545
