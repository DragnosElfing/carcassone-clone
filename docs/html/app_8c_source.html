153static void remove_last_utf8_char(char* str) {
154 if(str == NULL) return;
155 size_t len = strlen(str);
156 if (len == 0) return;
157
158 char* p = (char*)str + len - 1;
159
160 // UTF-8 enkódolási szabályok:
161 // - 1-byte karakter: 0xxxxxxx
162 // - 2-byte karakter: 110xxxxx 10xxxxxx
163 // - 3-byte karakter: 1110xxxx 10xxxxxx 10xxxxxx
164 // - 4-byte karakter: 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
165
166 while (p >= (char*)str) {
167 // Megkeressük az első, nem "continuation" bitet; odatesszük a \0-t.
168 if ((*p & 0b11000000) != 0b10000000) {
169 *p = '\0';
170 break;
171 }
172 --p;
173 }
174}

176size_t get_utf8_length(char* str)
177{
178 if(str == NULL) return 0U;
179
180 char* p = str;
181 size_t len = 0U;
182 while(*p != '\0') {
183 if ((*p & 0b11000000) != 0b10000000) {
184 ++len;
185 }
186 ++p;
187 }
188
189 return len;
190}
